
create  table #tempd
    (
      codConcepto int ,
      mes int ,
      anio int ,
      estado int
    )


declare @an int 
declare @mes int 
set @an = 2015
set @mes = 0
while @an <= datePart(year, getDate())
    and @mes <= 12
    begin
      
        set @mes = @mes + 1		
        
       
			
		
        insert  into #tempd
                ( codConcepto, mes, anio, estado )
        values  ( 1,
					 -- mes - int
                  @mes, -- anio - int
                  @an, 0  -- estado - int
                  )
            
			
        if @mes = 12
            begin
                set @an = @an + 1
                set @mes = 0
            end
			 
    end

insert  #tempd
        ( codConcepto ,
          mes ,
          anio ,
          estado
        )
        select  2 ,
                mes ,
                anio ,
                estado
        from    #tempd
insert  #tempd
        ( codConcepto ,
          mes ,
          anio ,
          estado
        )
        select  3 ,
                mes ,
                anio ,
                estado
        from    #tempd where codConcepto =1
