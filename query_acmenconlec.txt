

create  table #tempd
    (
      mes int ,
      anio int ,
      estado int
    )


declare @an int 
declare @mes int 

set @an = 2015
set @mes = 0
while @an <= datePart(year, getDate()) and @mes <= 12
    begin
      
        set @mes = @mes + 1		
        
            
        insert  into #tempd
                ( mes, anio, estado )
        values  ( @an, -- mes - int
                  @mes, -- anio - int
                  0  -- estado - int
                  )
		if @mes = 12
			BEGIN
			set @an = @an + 1
			set @mes = 0
			END
    end
