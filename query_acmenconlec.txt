
create  table #tempd
    (
      codConcepto int ,
      mes int ,
      anio int ,
      estado int,
	  monto decimal (18,2)
	  
    )


declare @an int 
declare @mes int 
declare @concepto int
set @an = 2015
set @mes = 1
set @concepto = 1
while @an <= datePart(year, getDate())  --and @mes <= 12
    begin

	 while @concepto <=3
	  BEGIN
	  	
	  
      insert #tempd
              ( codConcepto, mes, anio, estado, monto )
      values  ( @concepto, -- codConcepto - int
                @mes, -- mes - int
                @an, -- anio - int
                0, -- estado - int
                0  -- monto - decimal
                )  
		 set @concepto =  @concepto +1
	   end
       
	   set @concepto = 1

	  declare @var int 
	   set @var = @mes
	   set @mes = @mes +1
	   
	   if @var =12
	   BEGIN
	   	 set @an = @an +1
		 set @mes = 1
	   END
			 
    end
